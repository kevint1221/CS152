function main;
beginparams
endparams

beginlocals

  n : integer;
  i, j, k: integer;
  t : array [20] of integer;

endlocals

beginbody	## main program

  read i, j;

  k := 0;
  n := 20;
  
  do 
    beginloop
       t[k] := k;
       k := k + 1;
       write t[k-1];
    endloop
while k < n;
endbody
